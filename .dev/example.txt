# @shgen external _shgen_test_get_current_context() { kubectl config current-context 2>/dev/null || echo "none"; }

# Passthrough completion example:
# sh-gen-test passes unrecognised commands through to kubectl. The external
# block injects a function that loads kubectl's own bash completion handler
# and delegates to it, so all standard kubectl commands remain completable.
# @shgen external _shgen_test_kubectl_passthrough() { type __start_kubectl &>/dev/null || source <(kubectl completion bash 2>/dev/null); __start_kubectl; }

# The passthrough function is wired as a validation so it can be referenced
# from complete= on a positional argument with no flag name.
# @shgen validation kubectl-passthrough _shgen_test_kubectl_passthrough

# -----------------------------------------------------------------------------
# Root-level global flags (attached to the root module)

# --context selects the cluster context; completes from the contexts validation
# @shgen argument parent=sh-gen-test complete=contexts --context Select the cluster context to use

# --namespace / -n scopes all operations to a specific namespace
# @shgen argument parent=sh-gen-test complete=namespaces --namespace Kubernetes namespace to target (default: default)

# --output / -o controls output format for commands that produce structured data
# @shgen argument parent=sh-gen-test complete=output-formats --output Output format: json, yaml, table, wide, template

# --log-level sets global verbosity
# @shgen argument parent=sh-gen-test complete=log-levels --log-level Set the log level (debug, info, warn, error)

# --config points to an alternative config file; file completion is most useful here
# @shgen argument parent=sh-gen-test complete=file --config Path to an alternative config file

# --verbose enables verbose/debug output (boolean flag, no value — complete=none)
# @shgen argument parent=sh-gen-test complete=none --verbose Enable verbose output

# --dry-run prints what would happen without making changes
# @shgen argument parent=sh-gen-test complete=none --dry-run Print actions without executing them

# --region sets the cloud provider region for resource operations
# @shgen argument parent=sh-gen-test complete=regions --region Cloud provider region override

# --- get -----------------------------------------------------------------------
# Retrieve one or more resources from the cluster
# @shgen command parent=sh-gen-test get Retrieve one or more platform resources

# Arguments for 'get'
# @shgen argument parent=get --selector Filter resources by label selector (key=value pairs)
# @shgen argument parent=get complete=output-formats --output Output format override for this command
# @shgen argument parent=get complete=none --watch Stream live updates to the resource list
# @shgen argument parent=get complete=namespaces --namespace Namespace scope for this get operation
# @shgen argument parent=get complete=none --show-labels Include labels in the output table
# @shgen argument parent=get complete=none --all-namespaces List resources across all namespaces

# --- describe ------------------------------------------------------------------
# Show detailed information about a specific resource
# @shgen command parent=sh-gen-test describe Show detailed information about a resource

# Arguments for 'describe'
# @shgen argument parent=describe complete=pod-names --name Name of the resource to describe
# @shgen argument parent=describe complete=namespaces --namespace Namespace of the resource
# @shgen argument parent=describe complete=none --events Include recent events in output
# @shgen argument parent=describe complete=none --show-managed-fields Show managed field metadata

# --- apply ---------------------------------------------------------------------
# Apply a configuration manifest to the cluster from a file or directory
# @shgen command parent=sh-gen-test apply Apply a configuration manifest to the cluster

# Arguments for 'apply'
# @shgen argument parent=apply complete=file --filename Path to the manifest file or directory to apply
# @shgen argument parent=apply complete=none --recursive Recursively process manifest files in directories
# @shgen argument parent=apply complete=none --force Force re-apply even when no diff is detected
# @shgen argument parent=apply complete=none --prune Delete resources not present in the manifest
# @shgen argument parent=apply complete=output-formats --output Output format for the operation result
# @shgen argument parent=apply complete=none --server-side Use server-side apply strategy
# @shgen argument parent=apply complete=none --validate Run manifest validation before applying

# --- delete --------------------------------------------------------------------
# Remove one or more resources from the cluster
# @shgen command parent=sh-gen-test delete Remove one or more platform resources

# Arguments for 'delete'
# @shgen argument parent=delete complete=pod-names --name Name of the resource to delete
# @shgen argument parent=delete complete=namespaces --namespace Namespace of the resource to delete
# @shgen argument parent=delete complete=none --force Force immediate deletion bypassing graceful termination
# @shgen argument parent=delete complete=durations --grace-period Override the grace period seconds before deletion
# @shgen argument parent=delete complete=bool-values --cascade Also delete dependent resources (default: true)
# @shgen argument parent=delete complete=none --wait Wait until the resource is fully deleted before returning

# --- logs ----------------------------------------------------------------------
# Stream or retrieve logs from a running container
# @shgen command parent=sh-gen-test logs Stream or retrieve logs from a running container

# Arguments for 'logs'
# @shgen argument parent=logs complete=pod-names --pod Name of the pod to retrieve logs from
# @shgen argument parent=logs complete=none --container Name of the container within the pod
# @shgen argument parent=logs complete=none --follow Stream logs in real-time (tail -f equivalent)
# @shgen argument parent=logs complete=none --previous Retrieve logs from the previously terminated container
# @shgen argument parent=logs complete=tail-lines --tail Number of recent log lines to show (default: -1 = all)
# @shgen argument parent=logs complete=durations --since Only return logs newer than a relative duration e.g. 5m
# @shgen argument parent=logs complete=log-levels --level Minimum log level to include in output
# @shgen argument parent=logs complete=file --output-file Write logs to a local file instead of stdout

# @shgen argument parent=exec complete=none --tty Allocate a TTY for the session
# @shgen argument parent=exec complete=none -- Command separator: everything after -- is the command
